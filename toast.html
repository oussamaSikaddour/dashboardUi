Toast Notifications

Toast Notification Component

The Toast component provides lightweight, temporary notifications that appear to inform users of success, warnings, or general system messages. Toasts are meant to be unobtrusive, automatically dismissible, and accessible.

The component is built using ARIA roles for screen reader compatibility and includes customizable modifier classes to define its purpose (such as .info or .warning).

  • .toast__container — the main wrapper for the toast notification. It handles the toast’s visibility, styling, and accessibility context.
  • .toast__closer — a close button that allows users to manually dismiss the notification. Typically styled as a small “×” (close) icon made from two spans.
  • .toast — the message area inside the container, containing the notification text.
  • .toast__opener — button used to trigger or simulate a toast event, such as after a form submission or an action.
  • .info — modifier class applied to .toast__container to indicate an informational message (e.g., success, neutral updates).
  • .warning — modifier class applied to .toast__container to indicate warnings or alerts (e.g., caution messages).

Accessibility Notes:

  • role="status" is used to announce toast content changes without interrupting screen readers.
  • aria-labelledby="toast_label" links the container to its descriptive heading for context.
  • The close button includes an aria-label for clear purpose even if the icon is purely visual.
  • For more information about JavaScript integration, visit the documentation [here].

Usage Tips:

  • Use .info for general updates or success confirmations.
  • Use .warning for cautionary or attention-grabbing messages.
  • Toasts should disappear automatically after a short delay or when closed manually.
  • Avoid using toasts for critical, blocking errors — use the Errors Handler component instead.

Example markup (Info Toast):



  <button class="button toast__opener button--info">
    Send Toast Event
  </button>

  <div
      role="status"
      aria-labelledby="toast_label"
      class="toast__container info"
      id="defaultToast"
  >
      <h2 id="toast_label" class="sr-only">Success Notification</h2>

      <button
          class="toast__closer"
          aria-label="Close notification"
      >
          <span></span>
          <span></span>
      </button>

      <div class="toast">
          You have succeeded in completing the operation.
      </div>
  </div>

JS:


      
import { setAriaAttributes } from "../utils/Aria"; // Accessibility helper
import { dispatchCustomEvent } from "../utils/DespatchCustomEvent";

// Toast State Management

const getToastState = (toast) => {

  return toast.classList.contains("open");

};

const closeToast = (toast) => {

  toast.classList.remove("open");
  setAriaAttributes(true, "-1", toast);

};

const openToast = (toast) => {

  toast.classList.add("open");
  setAriaAttributes(false, "0", toast);
  toast.focus(); // Bring focus for accessibility
  setTimeout(() => {

    closeToast(toast); 

  }, 30000); 

};

// Event Handlers

const handleOpenToastClick = () => {

  dispatchCustomEvent("show-toast");

};

const handleToastClick = (toast) => {

  return () => closeToast(toast);

};

const handleToastKeyDown = (toast) => {

  return (e) => {

    if (e.key === "Enter" || e.key === " ") {

      closeToast(toast);

    }

  };

};

const handleShowToastEvent = (toast) => {

  return (e) => {

    e.preventDefault();

    openToast(toast);

  };

};

// Element Selection
const getToastElements = () => {

  const openButton = document.querySelector(".toast__opener");

  const closeButton = document.querySelector(".toast__closer");

  const toast = document.querySelector(".toast__container");

  
  return { openButton, closeButton, toast };
};

const validateToastElements = (openButton, closeButton, toast) => {

  if (!toast) {

    console.warn('Toast container element not found');

    return false;

  }
  
  if (!openButton) {

    console.warn('Toast opener element not found');

    return false;

  }
  
  if (!closeButton) {

    console.warn('Toast closer element not found');

    return false;

  }
  
  return true;

};

// Event Listener Setup
const setupToastEventListeners = (openButton, closeButton, toast) => {

  const openHandler = handleOpenToastClick;

  const clickHandler = handleToastClick(toast);

  const keyDownHandler = handleToastKeyDown(toast);

  const customEventHandler = handleShowToastEvent(toast);

  
  openButton.addEventListener("click", openHandler);

  closeButton.addEventListener("click", clickHandler);

  toast.addEventListener("click", clickHandler);

  toast.addEventListener("keydown", keyDownHandler);

  document.addEventListener('show-toast', customEventHandler);

};

// Main Toast Component

const Toast = () => {

  const { openButton, closeButton, toast } = getToastElements();

  
  if (!validateToastElements(openButton, closeButton, toast)) {

    return;

  }
  
  setupToastEventListeners(openButton, closeButton, toast);
  
};

export default Toast;
